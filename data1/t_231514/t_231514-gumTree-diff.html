        view = new PriorityView( this );
        
    
    }

    }

        }
    }
    
    public void itemStateChanged(ItemEvent e) {
if (e.getStateChange() == ItemEvent.SELECTED) {
            controller.getModel().setPriority( (String) view.getSelectedItem() );

}
